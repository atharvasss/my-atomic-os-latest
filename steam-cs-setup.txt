====================
Fedora Silverblue / Atomic
Steam (Flatpak)
====================

# Update the system and reboot
rpm-ostree upgrade
systemctl reboot

# Enable Flathub (needed for Steam Flatpak)
flatpak remote-add --if-not-exists flathub \
https://flathub.org/repo/flathub.flatpakrepo

# Install Steam (Flatpak ONLY)
flatpak install -y flathub com.valvesoftware.Steam

# Install Steam runtimes and Proton
# Fixes Steamworks SDK / SteamClient errors
flatpak install -y flathub \
com.valvesoftware.Steam.CompatibilityTool.SteamLinuxRuntime \
com.valvesoftware.Steam.CompatibilityTool.Proton

# Allow Steam access to GPU, controllers, and files
flatpak override --user \
--filesystem=home \
--filesystem=xdg-data/Steam \
--device=all \
--share=network \
com.valvesoftware.Steam

# Launch Steam and let it update
flatpak run com.valvesoftware.Steam

# Inside Steam (MANUAL):
# Settings -> Compatibility -> Enable Steam Play for all titles
# Settings -> Shader Pre-Caching -> Enable
